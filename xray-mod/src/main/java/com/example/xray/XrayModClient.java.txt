package com.example.xray;

import com.mojang.blaze3d.systems.RenderSystem;
import net.fabricmc.api.ClientModInitializer;
import net.fabricmc.fabric.api.client.keybinding.v1.KeyBindingHelper;
import net.fabricmc.fabric.api.client.rendering.v1.WorldRenderEvents;
import net.fabricmc.fabric.api.client.event.lifecycle.v1.ClientTickEvents;

import net.minecraft.block.Block;
import net.minecraft.block.Blocks;
import net.minecraft.client.MinecraftClient;
import net.minecraft.client.option.KeyBinding;
import net.minecraft.client.render.*;
import net.minecraft.client.util.InputUtil;
import net.minecraft.client.util.math.MatrixStack;
import net.minecraft.util.math.BlockPos;

import org.lwjgl.glfw.GLFW;

import java.util.HashSet;
import java.util.Set;

public class XrayModClient implements ClientModInitializer {

    private static KeyBinding toggleKey;
    private static boolean enabled = false;
    private static final int RADIUS = 32;

    private static final Set<Block> ORES = new HashSet<>();

    static {
        ORES.add(Blocks.COAL_ORE);
        ORES.add(Blocks.IRON_ORE);
        ORES.add(Blocks.GOLD_ORE);
        ORES.add(Blocks.REDSTONE_ORE);
        ORES.add(Blocks.LAPIS_ORE);
        ORES.add(Blocks.DIAMOND_ORE);
        ORES.add(Blocks.EMERALD_ORE);
    }

    @Override
    public void onInitializeClient() {

        toggleKey = KeyBindingHelper.registerKeyBinding(new KeyBinding(
                "key.xray.toggle",
                InputUtil.Type.KEYSYM,
                GLFW.GLFW_KEY_K,
                "category.xray"
        ));

        ClientTickEvents.END_CLIENT_TICK.register(client -> {
            while (toggleKey.wasPressed()) {
                enabled = !enabled;
            }
        });

        WorldRenderEvents.AFTER_ENTITIES.register(context -> {
            if (!enabled) return;

            MinecraftClient client = MinecraftClient.getInstance();
            if (client.world == null || client.player == null) return;

            MatrixStack matrices = context.matrixStack();
            VertexConsumerProvider.Immediate consumers = context.consumers();

            double camX = client.gameRenderer.getCamera().getPos().x;
            double camY = client.gameRenderer.getCamera().getPos().y;
            double camZ = client.gameRenderer.getCamera().getPos().z;

            RenderSystem.disableTexture();
            RenderSystem.enableBlend();
            RenderSystem.defaultBlendFunc();
            RenderSystem.lineWidth(2.0F);

            VertexConsumer vc = consumers.getBuffer(RenderLayer.getLines());

            BlockPos playerPos = client.player.getBlockPos();

            for (int dx = -RADIUS; dx <= RADIUS; dx++) {
                for (int dy = -RADIUS; dy <= RADIUS; dy++) {
                    for (int dz = -RADIUS; dz <= RADIUS; dz++) {

                        BlockPos pos = playerPos.add(dx, dy, dz);
                        Block block = client.world.getBlockState(pos).getBlock();

                        if (ORES.contains(block)) {
                            float r = 0f, g = 1f, b = 1f;

                            if (block == Blocks.DIAMOND_ORE) { r = 0f; g = 0.8f; b = 1f; }
                            if (block == Blocks.GOLD_ORE)    { r = 1f; g = 0.85f; b = 0f; }
                            if (block == Blocks.IRON_ORE)    { r = 0.9f; g = 0.9f; b = 0.9f; }
                            if (block == Blocks.COAL_ORE)    { r = 0.1f; g = 0.1f; b = 0.1f; }
                            if (block == Blocks.REDSTONE_ORE){ r = 1f; g = 0.2f; b = 0.2f; }
                            if (block == Blocks.LAPIS_ORE)   { r = 0.2f; g = 0.2f; b = 1f; }
                            if (block == Blocks.EMERALD_ORE) { r = 0.2f; g = 1f; b = 0.4f; }

                            drawBox(vc, matrices, pos, r, g, b, 0.6f, camX, camY, camZ);
                        }
                    }
                }
            }

            consumers.draw();
            RenderSystem.enableTexture();
            RenderSystem.disableBlend();
        });
    }

    private void drawBox(VertexConsumer vc, MatrixStack matrices, BlockPos pos,
                         float r, float g, float b, float alpha,
                         double camX, double camY, double camZ) {

        double x = pos.getX() - camX;
        double y = pos.getY() - camY;
        double z = pos.getZ() - camZ;

        MatrixStack.Entry entry = matrices.peek();
        Matrix4f model = entry.getModel();

        putLine(vc, model, x, y, z, x+1, y, z, r, g, b, alpha);
        putLine(vc, model, x, y, z, x, y+1, z, r, g, b, alpha);
        putLine(vc, model, x, y, z, x, y, z+1, r, g, b, alpha);

        putLine(vc, model, x+1, y+1, z+1, x, y+1, z+1, r, g, b, alpha);
        putLine(vc, model, x+1, y+1, z+1, x+1, y, z+1, r, g, b, alpha);
        putLine(vc, model, x+1, y+1, z+1, x+1, y+1, z, r, g, b, alpha);

        putLine(vc, model, x+1, y, z, x+1, y, z+1, r, g, b, alpha);
        putLine(vc, model, x, y+1, z, x, y+1, z+1, r, g, b, alpha);

        putLine(vc, model, x, y, z+1, x+1, y, z+1, r, g, b, alpha);
        putLine(vc, model, x, y+1, z+1, x+1, y+1, z+1, r, g, b, alpha);

        putLine(vc, model, x+1, y, z, x+1, y+1, z, r, g, b, alpha);
        putLine(vc, model, x, y, z, x, y, z+1, r, g, b, alpha);
    }

    private void putLine(VertexConsumer vc, Matrix4f model,
                         double x1, double y1, double z1,
                         double x2, double y2, double z2,
                         float r, float g, float b, float a) {

        vc.vertex(model, (float)x1, (float)y1, (float)z1).color(r, g, b, a).next();
        vc.vertex(model, (float)x2, (float)y2, (float)z2).color(r, g, b, a).next();
    }
}
